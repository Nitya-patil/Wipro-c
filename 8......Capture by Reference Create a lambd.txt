Capture by Reference: Create a lambda that captures a string by reference, appends a fixed prefix, and returns the modified string.


#include<iostream>
#include<string>
using namespace std;

int main(){
    string str = "Soni";
    
    auto addpre = [&str]()-> string {return "Gourav " + str;};
    
    string modifiedStr = addpre();
    
    std::cout << "New string is : " << modifiedStr << std::endl;
    
}
